To auto-start Tomcat Server on reboot

Create tomcat.sh under /etc/init.d/ directory

vi /etc/init.d/tomcat.sh
OR
gedit /etc/init.d/tomcat.sh

Add the following [Note: JAVA_HOME and Tomcat path]

#!/bin/sh
# chkconfig: 2345 80 20
# Description: Tomcat Server Start/Shutdown script
export JAVA_HOME=/opt/jdk1.8.0_91
case $1 in
start)
cd /opt/apache-tomcat-8.0.33/bin/
./startup.sh
;;
stop)
cd /opt/apache-tomcat-8.0.33/bin/
./shutdown.sh
;;
restart)
cd /opt/apache-tomcat-8.0.33/bin/
./shutdown.sh
cd /opt/apache-tomcat-8.0.33/bin/
./startup.sh
;;
esac
exit 0

Save tomcat.sh
